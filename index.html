责任链模式
STUDY 2024/9/12

重要程度：3
责任链模式是一种行为设计模式，它允许多个对象有机会处理请求，从而避免请求的发送者和接受者之间的耦合关系。这个模式将这些对象连接成一条链，并沿着这条链传递请求，直到有一个对象处理它为止。

  1. 主要组成
    1. 抽象处理者
    2. 具体处理者
    3. 客户端
  2. 典型应用：日志系统。定义一个不同级别的日志信息，如调试、信息、警告、错误等。
  3. 优点：
    1. 降低耦合度：请求发送者和请求接收者解耦，客户端不需要了解处理请求的具体对象。
    2. 增强灵活性：可以通过添加或删除处理者来修改链的结构。
    3. 职责单一原则：每个处理者只专注于自己能够处理的请求。
  4. 缺点：
    1. 请求未必被处理：如果链尾的处理者也无法处理请求，那么请求可能会被丢弃。
    2. 调试复杂：链的结构在运行时才确定，如果链过长，可能会增加调试的难度。
  5. 代码
#include <iostream>
#include <string>

// 抽象处理者类
class Logger {
protected:
    int level;
    Logger* nextLogger;

public:
    Logger(int lvl) : level(lvl), nextLogger(nullptr) {}

    void setNextLogger(Logger* next) {
        nextLogger = next;
    }

    void logMessage(int lvl, const std::string& message) {
        if (this->level <= lvl) {
            write(message);
        }
        if (nextLogger != nullptr) {
            nextLogger->logMessage(lvl, message);
        }
    }

    virtual void write(const std::string& message) = 0; // 纯虚函数，子类必须实现
};

// 具体处理者类：DebugLogger
class DebugLogger : public Logger {
public:
    DebugLogger(int lvl) : Logger(lvl) {}

    void write(const std::string& message) override {
        std::cout << "Debug Logger: " << message << std::endl;
    }
};

// 具体处理者类：InfoLogger
class InfoLogger : public Logger {
public:
    InfoLogger(int lvl) : Logger(lvl) {}

    void write(const std::string& message) override {
        std::cout << "Info Logger: " << message << std::endl;
    }
};

// 具体处理者类：WarningLogger
class WarningLogger : public Logger {
public:
    WarningLogger(int lvl) : Logger(lvl) {}

    void write(const std::string& message) override {
        std::cout << "Warning Logger: " << message << std::endl;
    }
};

// 具体处理者类：ErrorLogger
class ErrorLogger : public Logger {
public:
    ErrorLogger(int lvl) : Logger(lvl) {}

    void write(const std::string& message) override {
        std::cout << "Error Logger: " << message << std::endl;
    }
};

// 客户端代码
int main() {
    // 创建具体的日志处理者
    Logger* errorLogger = new ErrorLogger(3);
    Logger* warningLogger = new WarningLogger(2);
    Logger* infoLogger = new InfoLogger(1);
    Logger* debugLogger = new DebugLogger(0);

    // 设置责任链：Debug -> Info -> Warning -> Error
    debugLogger->setNextLogger(infoLogger);
    infoLogger->setNextLogger(warningLogger);
    warningLogger->setNextLogger(errorLogger);

    // 测试责任链
    std::cout << "Sending a debug level message:" << std::endl;
    debugLogger->logMessage(0, "This is a debug message.");

    std::cout << "Sending an info level message:" << std::endl;
    debugLogger->logMessage(1, "This is an info message.");

    std::cout << "Sending a warning level message:" << std::endl;
    debugLogger->logMessage(2, "This is a warning message.");

    std::cout << "Sending an error level message:" << std::endl;
    debugLogger->logMessage(3, "This is an error message.");

    // 释放内存
    delete debugLogger;
    delete infoLogger;
    delete warningLogger;
    delete errorLogger;

    return 0;
}

装饰者模式
STUDY 2024/9/12

重要性程度：3
动态地给一个对象增加一些额外的职责，就增加对象功能来说，装饰模式比生成子类实现更加灵活。是一种对象结构型模式。与继承关系相比，关联关系的主要优势在于不会破坏类的封装性，而且继承是一种耦合度较大的静态关系，无法在程序运行时动态扩展。

  1. 核心组件
    1. 抽象组件
    2. 具体组件
    3. 抽象饰品
    4. 具体饰品
  2. 优点：
    1. 灵活性高。装饰模式与继承关系的目的都是要扩展对象的功能，但是装饰模式可以提供比继承更多的灵活性。可以通过一种动态的方式来扩展一个对象的功能，通过配置文件可以在运行时选择不同的装饰器，从而实现不同的行为。
    2. 遵循开放封闭原则。具体构件类与具体装饰类可以独立变化，用户可以根据需要增加新的具体构件类和具体装饰类，在使用时再对其进行组合，原有代码无须改变，符合“开闭原则”
    3. 单一职责原则。
  3. 缺点：
    1. 增加复杂性
    2. 调试困难
  4. 代码
    抽象组件：咖啡
    具体组件：黑咖啡
    抽象饰品：小料
    具体饰品：牛奶 巧克力
#include <iostream>
#include <memory>
#include <string>
using namespace std;
// 抽象咖啡
class Coffee {
public:
    virtual string getDescription() const {
        return "unknown coffee";
    }
    virtual double cost() const = 0;
    virtual ~Coffee() = default;
};

// 具体组件：黑咖啡
class BlackCoffee : public Coffee {
public:
    std::string getDescription() const override {
        return "Black Coffee";
    }

    double cost() const override {
        return 2.0;
    }
};

// 抽象饰品
class CoffeeDecorator : public Coffee {
protected:
    unique_ptr<Coffee> coffee;//被装饰对象
public:
    CoffeeDecorator(unique_ptr<Coffee> c) : coffee(std::move(c)) {};
};

// 具体装饰者：奶油
class CreamDecorator : public CoffeeDecorator {
public:
    CreamDecorator(std::unique_ptr<Coffee> c) : CoffeeDecorator(std::move(c)) {}

    std::string getDescription() const override {
        return coffee->getDescription() + ", Cream";
    }

    double cost() const override {
        return coffee->cost() + 0.5;
    }
};

// 具体装饰者：巧克力
class ChocolateDecorator : public CoffeeDecorator {
public:
    ChocolateDecorator(std::unique_ptr<Coffee> c) : CoffeeDecorator(std::move(c)) {}

    std::string getDescription() const override {
        return coffee->getDescription() + ", Chocolate";
    }

    double cost() const override {
        return coffee->cost() + 0.7;
    }
};
int main() {
    // 创建一个黑咖啡
    std::unique_ptr<Coffee> myCoffee = std::make_unique<BlackCoffee>();
    std::cout << myCoffee->getDescription() << " 价格: $" << myCoffee->cost() << std::endl;

    // 在黑咖啡上加奶油
    myCoffee = std::make_unique<CreamDecorator>(std::move(myCoffee));
    std::cout << myCoffee->getDescription() << " 价格: $" << myCoffee->cost() << std::endl;

    // 在加了奶油的咖啡上再加巧克力
    myCoffee = std::make_unique<ChocolateDecorator>(std::move(myCoffee));
    std::cout << myCoffee->getDescription() << " 价格: $" << myCoffee->cost() << std::endl;

    return 0;
}

理解：
与其说是装饰，不如理解成，一个饰品克隆了一个被装饰者，但是克隆的时候，加上了饰品。但是克隆一个肉体太麻烦，就使用移动语义仅仅是抹除了思想上的本我，然后加上饰品创造了一个新我。
以此类推。

策略模式
STUDY 2024/9/12

策略模式定义一系列算法，将每一个算法封装起来，并让它们可以相互替换。策略模式让算法独立于使用它的客户而变化，也称为政策模式。策略模式是一种对象行为型模式。
重要性程度：4

  1. 主要组成
    1. 环境类
    2. 抽象策略类
    3. 具体策略类
      根据环境的某些特质，采用不同的策略。
  2. 优点
    1. 完美支持“开闭原则”，用户可以在不修改原有系统的基础上选择算法或行为，也可以灵活地增加新的算法或行为。
    2. 策略模式提供了可以替换继承关系的办法
    3. 使用策略模式可以避免提供使用多重条件转移语句
  3. 缺点
    1. 客户端必须知道是所有的策略类，并自行决定使用哪一个策略类。
    2. 策略模式将造成产生很多策略类，可以通过使用享元模式在一定程度上减少对象的数量。
  4. 代码
    支付策略：分为现金支付、手机支付
#include <iostream>
#include <memory>

// 抽象策略类
class PaymentStrategy {
public:
    virtual void pay(int amount) = 0;
    virtual ~PaymentStrategy() = default;
};

// 具体策略类
class CreaditCardPayment : public PaymentStrategy {
public:
    void pay(int amount) override {
        std::cout << "使用信用卡支付了" << amount << "元。" << std::endl;
    }
};

class CashPayment : public PaymentStrategy {
public:
    void pay(int amount) override {
        std::cout << "使用现金支付了" << amount << "元。" << std::endl;
    }
};

// 环境类
class PaymentContext {
private:
    std::unique_ptr<PaymentStrategy> _strategy;
public:
    void setPaymentStrategy(std::unique_ptr<PaymentStrategy> strategy) {
        _strategy = std::move(strategy);
    }
    void executePayment(int amount) {
        if (_strategy) {
            _strategy->pay(amount);
        }
        else {
            std::cout << "未指定支付方式。" << std::endl;
        }
    }
};

int main() {
    PaymentContext context;
    context.setPaymentStrategy(std::make_unique<CreaditCardPayment>());
    context.executePayment(100);
    context.setPaymentStrategy(std::make_unique<CashPayment>());
    context.executePayment(500);
    return 0;
}

观察者模式
STUDY 2024/9/12

观察者模式定义对象间的一种一对多依赖关系，使得每当一个对象状态发生改变时，其相关依赖对象皆得到通知并被自动更新。观察者模式又叫做发布-订阅模式。是一种对象行为模式。
重要性程度：5

  1. 主要组成
    1. 抽象观察者
    2. 具体观察者
    3. 抽象目标
    4. 具体目标
  2. 优点
    1. 可以实现表示层和数据逻辑层的分离，并定义了稳定的消息更新传递机制，抽象了更新接口，使得可以有各种各样不同的表示层作为具体观察者角色。
    2. 观察者模式在观察目标和观察者之间建立一个抽象的耦合。
    3. 观察者模式支持广播通信。
    4. 观察者模式符合开闭原则
  3. 缺点
    1. 通知机制耗费时间
    2. 如果存在循环依赖，会导致出现循环调用，可能导致系统崩溃。
  4. 代码
#include <memory>
#include <iostream>
#include <vector>

class Subject;
// 抽象观察者
class Observer : public std::enable_shared_from_this<Observer> {
public:
    Observer(const std::string& name) :_name(name) {};
    virtual ~Observer() = default;
    virtual void update(const std::string& message) = 0;
    virtual void subscribe(std::shared_ptr<Subject> subject);
    virtual void unsubscribe(std::shared_ptr<Subject> subject);
protected:
    std::string _name;
    std::vector<std::shared_ptr<Subject>> subjects;
};

// 具体观察者
class ConcreteObserver : public Observer {
public:
    ConcreteObserver(const std::string& name) : Observer(name) {}

    void update(const std::string& message) override {
        std::cout << _name << " received message: " << message << std::endl;
    }
};
// 抽象目标
class Subject {
public:
    virtual ~Subject() = default;
    virtual void attach(std::shared_ptr<Observer> observer) = 0;
    virtual void detach(std::shared_ptr<Observer> observer) = 0;
    virtual void notify() = 0;
};
// 具体目标
class ConcreteSubject : public Subject {
private:
    std::vector<std::shared_ptr<Observer>> observers;
    std::string _message;
public:
public:
    void attach(std::shared_ptr<Observer> observer) override {
        observers.push_back(observer);
    }

    void detach(std::shared_ptr<Observer> observer) override {
        observers.erase(std::remove(observers.begin(), observers.end(), observer), observers.end());
    }

    void notify() override {
        for (auto& observer : observers) {
            observer->update(_message);  // 通知每个观察者
        }
    }

    void createMessage(const std::string& message) {
        _message = message;
        notify();  // 当状态改变时，通知所有观察者
    }
};

void Observer::subscribe(std::shared_ptr<Subject> subject) {
    subjects.push_back(subject);
    subject->attach(shared_from_this());
}
void Observer::unsubscribe(std::shared_ptr<Subject> subject) {
    subjects.erase(std::remove(subjects.begin(), subjects.end(), subject), subjects.end());
    subject->detach(shared_from_this());
}

// 测试观察者模式
int main() {
    std::shared_ptr<ConcreteSubject> subject = std::make_shared<ConcreteSubject>();

    std::shared_ptr<Observer> observer1 = std::make_shared<ConcreteObserver>("Observer 1");
    std::shared_ptr<Observer> observer2 = std::make_shared<ConcreteObserver>("Observer 2");
    std::shared_ptr<Observer> observer3 = std::make_shared<ConcreteObserver>("Observer 3");

    observer1->subscribe(subject);
    observer2->subscribe(subject);
    observer3->subscribe(subject);

    subject->createMessage("Hello, Observers!");  // 向所有观察者发送消息

    observer2->unsubscribe(subject);

    subject->createMessage("Second message!");  // 发送另一条消息，只有剩下的观察者会接收到

    return 0;
}

简单工厂、工厂方法与抽象工厂
STUDY 2024/9/12

简单工厂模式、工厂方法模式、抽象工厂模式均属于创建型模式。

简单工厂模式

  1. 主要组成
    1. 工厂类：负责实现创建所有实例的内部逻辑，通常提供一个静态方法，根据传入的参数类型来决定初始化哪种具体类的对象。
    2. 抽象产品类：定义了产品的公共接口。
    3. 具体产品类
  2. 优点
    1. 封装对象创建。工厂类含有必要的判断逻辑，可以决定在什么时候创建哪一个产品类的实例，客户端可以免除直接创建产品对象的责任，而仅仅“消费产品”；同时客户端无需知道所创建的具体产品类的类名，只需要知道具体产品类所对应的参数即可。（是否关心，主要看main函数创建时，是否回根据用户的目的进行工厂类、创建方法、返回对象的选择）
    2. 可扩展性：通过引入配置文件，可以在不修改任何客户端代码的情况先更换和增加新的具体产品类。
  3. 缺点
    1. 不符合开放封闭原则：每增加一种产品，都要修改工厂类的实现。
    2. 不符合单一职责原则：工厂类集中了所有对象的创建逻辑，职责较重，导致维护和管理困难。
  4. 代码
    生产果汁，客户端告诉工厂是什么水果，工厂负责生产果汁。
#include <iostream>
#include <memory>
#include <string>

// 抽象产品类-果汁
class Juice {
public:
    Juice(int sweetness) : _sweetness(sweetness) {};
    virtual ~Juice() = default;
    virtual void showLabel() = 0;
protected:
    int _sweetness;     // 甜度
};

// 具体产品类，苹果果汁、橙汁
class AppleJuice : public Juice {
public:
    AppleJuice(int sweetness) : Juice(sweetness) {}
    virtual void showLabel() override {
        std::cout << "This is apple juice, its sweetness is " << _sweetness << "." << std::endl;
    }
};

class OrangeJuice : public Juice {
public:
    OrangeJuice(int sweetness) : Juice(sweetness) {}
    virtual void showLabel() override {
        std::cout << "This is orange juice, its sweetness is " << _sweetness << "." << std::endl;
    }
};

// 工厂类
class JuiceFactory {
public:
    static std::unique_ptr<Juice> product(const std::string& fruit) {
        if (fruit == "apple") {
            return std::make_unique<AppleJuice>(10);
        }
        else if (fruit == "orange") {
            return std::make_unique<OrangeJuice>(20);
        }
        else return nullptr;
    }
};

int main() {
    auto product_0 = JuiceFactory::product("orange");
    auto product_1 = JuiceFactory::product("apple");
    if (product_0) {
        product_0->showLabel();
    }
    if (product_1) {
        product_1->showLabel();
    }
    return 0;
}

工厂方法模式

和简单工厂模式不同，为了解决简单工厂模式中工厂的职责复杂且不符合开放封闭原则。工厂方法模式将工厂类设置为一个抽象工厂类和若干个具体工厂类，之后增加具体产品类的时候只需要增加一个对应的具体工厂类即可，不需要对抽象工厂类进行更改。更见符合单一职责原则和开放封闭原则。

  1. 主要组成
    1. 抽象产品
    2. 具体产品
    3. 抽象工厂
    4. 具体工厂
  2. 优点：
    1. 用户只需要关心所需产品对应的工厂，无需关心创建细节，无需知道具体产品类的类名。
    2. 基于工厂角色和产品角色的多态性设计时工厂方法模式的关键。
    3. 扩展性：在系统中加入新产品时，无须修改抽象工厂和抽象产品提供的接口，无须修改客户端，也无须修改其他的具体工厂和具体产品，而只要添加一个具体工厂和具体产品就可以了。
  3. 缺点
    1. 添加新产品时，需要编写新的具体产品类和具体工厂类，类的个数成对增加。
    2. 由于考虑到系统的可扩展性，需要引入抽象层，在客户端代码中均使用抽象层进行定义，增加了系统的抽象性和理解难度，且在实现时可能需要用到DOM、反射等技术，增加了系统的实现难度。
  4. 代码
#include <iostream>
#include <memory>  // 包含智能指针的头文件

class Juice {
public:
    Juice(int sweetness) : _sweetness(sweetness) {};
    virtual ~Juice() = default;
    virtual void showLabel() = 0;
protected:
    int _sweetness;     // 甜度
};

// 具体产品类，苹果果汁、橙汁
class AppleJuice : public Juice {
public:
    AppleJuice(int sweetness) : Juice(sweetness) {}
    virtual void showLabel() override {
        std::cout << "This is apple juice, its sweetness is " << _sweetness << "." << std::endl;
    }
};

class OrangeJuice : public Juice {
public:
    OrangeJuice(int sweetness) : Juice(sweetness) {}
    virtual void showLabel() override {
        std::cout << "This is orange juice, its sweetness is " << _sweetness << "." << std::endl;
    }
};

// 工厂类
class JuiceFactory {
public:
    virtual ~JuiceFactory() = default;
    virtual std::unique_ptr<Juice> product() = 0;  // 返回智能指针
};

// 具体工厂类
class AppleJuiceFactory : public JuiceFactory {
public:
    virtual std::unique_ptr<Juice> product() override {
        return std::make_unique<AppleJuice>(10);
    }
};

class OrangeJuiceFactory : public JuiceFactory {
public:
    virtual std::unique_ptr<Juice> product() override {
        return std::make_unique<OrangeJuice>(20);
    }
};

int main() {
    // 使用智能指针管理工厂的生命周期
    std::unique_ptr<JuiceFactory> apple_juice_factory = std::make_unique<AppleJuiceFactory>();
    std::unique_ptr<JuiceFactory> orange_juice_factory = std::make_unique<OrangeJuiceFactory>();

    // 使用智能指针管理产品的生命周期
    std::unique_ptr<Juice> product_0 = apple_juice_factory->product();
    std::unique_ptr<Juice> product_1 = orange_juice_factory->product();

    product_0->showLabel();
    product_1->showLabel();

    return 0;
}

抽象工厂模式

对于上面的果汁的例子，一瓶果汁可以看作是由果汁本体、瓶子、标签组成。苹果果汁和橙汁的果汁本体和标签不同，但是瓶子可能是完全相同的。为了增加代码复用，且降低一个产品内部组件之间的耦合性（比如使用新的瓶子，不用更改苹果汁或橙汁的整个类的实现）。使用抽象工厂模式可以实现上面的功能。抽象工厂模式提供一个创建一系列相关或相互依赖对象的接口，而无需指定它们具体的类。抽象工厂模式又称为Kit模式，属于创建型模式。

  1. 主要组成
    1. 抽象组件类
    2. 具体组件类
    3. 抽象产品类
    4. 具体产品类
    5. 抽象工厂类
    6. 具体工厂类
      虽然上面列出了这么的类，但是具体实现还要看是否需要对应的抽象基类。
  2. 优点
    1. 抽象工厂模式隔离了具体类的生成，使得客户端并不需要知道什么被创建。由于这种隔离，更换一个具体工厂就变得相对容易。所有的具体工厂都实现了抽象工厂定义的那些公共接口，因此只需改变具体工厂的实例，就可以在某种程度上改变整个软件系统的行为。另外，应用抽象工厂模式可以实现高内聚低耦合的设计目的。
    2. 增加新的具体工厂和产品组很方便，无需更改已有系统，符合开放封闭原则。
  3. 缺点
    1. 在添加新的产品对象时，难以扩展抽象工厂来生产新种类的产品，这是因为在抽象工厂角色中规定了所有可能被创建的产品集合，要支持新种类的产品就意味着要对该接口进行扩展，而这将涉及到对抽象工厂角色及其所有子类的修改，显然会带来较大的不便。
    2. 开闭原则的倾斜性（增加新的工厂和产品族容易，增加新的产品等级结构麻烦）。
  4. 代码
#include <iostream>
#include <memory>

// 抽象产品：标签
class Label {
public:
    virtual ~Label() = default;
    virtual void showLabel() = 0;
};

// 具体产品：红色标签
class RedLabel : public Label {
public:
    virtual void showLabel() override {
        std::cout << "The label is red." << std::endl;
    }
};

// 具体产品：黄色标签
class YellowLabel : public Label {
public:
    virtual void showLabel() override {
        std::cout << "The label is yellow." << std::endl;
    }
};

// 抽象产品：瓶子
class Bottle {
public: 
    virtual ~Bottle() = default;
    virtual void showBottle() = 0;
};

// 具体产品：大瓶子
class BigBottle : public Bottle{
public:
    void showBottle() override {
        std::cout << "The bottle's volume is 1000 ml." << std::endl;
    }
};

// 具体产品：小瓶子
class TinyBottle : public Bottle {
public:
    void showBottle() override {
        std::cout << "The bottle's volume is 350 ml." << std::endl;
    }
};

// 抽象产品：果汁
class Juice {
public:
    virtual ~Juice() = default;
    virtual void showInfo() = 0;
};

// 具体产品：苹果汁
class AppleJuice : public Juice {
private:
    int _sweetness;
    std::unique_ptr<Label> _label;
    std::unique_ptr<Bottle> _bottle;
public:
    AppleJuice(int sweetness, std::unique_ptr<Label> label, std::unique_ptr<Bottle> bottle)
        : _sweetness(sweetness), _label(std::move(label)), _bottle(std::move(bottle)) {}

    virtual void showInfo() override {
        std::cout << "This is apple juice, its sweetness is " << _sweetness << "." << std::endl;
        _label->showLabel();
        _bottle->showBottle();
    }
};

// 具体产品：橙汁
class OrangeJuice : public Juice {
private:
    int _sweetness;
    std::unique_ptr<Label> _label;
    std::unique_ptr<Bottle> _bottle;
public:
    OrangeJuice(int sweetness, std::unique_ptr<Label> label, std::unique_ptr<Bottle> bottle)
        : _sweetness(sweetness), _label(std::move(label)), _bottle(std::move(bottle)) {}

    virtual void showInfo() override {
        std::cout << "This is orange juice, its sweetness is " << _sweetness << "." << std::endl;
        _label->showLabel();
        _bottle->showBottle();
    }
};

// 抽象工厂类
class JuiceFactory {
public:
    virtual ~JuiceFactory() = default;
    virtual std::unique_ptr<Label> createLabel() = 0;
    virtual std::unique_ptr<Bottle> createBottle() = 0;
    virtual std::unique_ptr<Juice> product() = 0;
};

// 具体工厂类：生产小瓶红标签苹果汁
class AppleJuiceFactory_tiny_red : public JuiceFactory {
public:
    virtual std::unique_ptr<Label> createLabel() override {
        return std::make_unique<RedLabel>();
    }

    virtual std::unique_ptr<Bottle> createBottle() override {
        return std::make_unique<TinyBottle>();
    }

    virtual std::unique_ptr<Juice> product() override {
        return std::make_unique<AppleJuice>(10, createLabel(), createBottle());
    }
};

// 具体工厂类：生产大瓶黄标签橙汁
class OrangeJuiceFactory_yellow_big : public JuiceFactory {
public:
    virtual std::unique_ptr<Label> createLabel() override {
        return std::make_unique<YellowLabel>();
    }

    virtual std::unique_ptr<Bottle> createBottle() override {
        return std::make_unique<BigBottle>();
    }

    virtual std::unique_ptr<Juice> product() override {
        return std::make_unique<OrangeJuice>(20, createLabel(), createBottle());
    }
};

int main() {
    std::unique_ptr<JuiceFactory> apple_juice_factory = std::make_unique<AppleJuiceFactory_tiny_red>();
    std::unique_ptr<JuiceFactory> orange_juice_factory = std::make_unique<OrangeJuiceFactory_yellow_big>();

    std::unique_ptr<Juice> product_0 = apple_juice_factory->product();
    std::unique_ptr<Juice> product_1 = orange_juice_factory->product();

    product_0->showInfo();
    product_1->showInfo();

    return 0;
}
